Avoid label statements [CODSTA.BP.AULS]



DESCRIPTION 


This rule identifies labels in code. An error is reported 
for each occurrence.
SINCE 

v6.0
NOTES 

A complementary rule UC.AUL may be beneficial in finding
only label statements which are either invalid or unused.
BENEFITS 

Using labels could cause an illogical flow, which can lead to
erroneous code and could make the code difficult to read.
Code can always be rewritten in a way that does not use
labels.
EXAMPLE 

package examples.rules.codsta.bp;

public class AULS
{
    public static void main (String args[]) {
        int a = 0;
        int b = 0;
        while (a <= 10) {
            tag:  //VIOLATION
            {
                a++;
                b = 0;
                while (b <= 10) {
                    b++;
                    System.out.println(a * b);
                    if (a == b)
                        break tag;
                }
            }
        }
    }
}
REPAIR 

package examples.rules.codsta.bp;

public class AULSFixed
{
    public static void main (String args[]) {
      
        int a = 0;
        int b = 0;
        boolean flag= true;

            a++;
            b = 0;
            flag= true;
            while ((b <= 10) && (flag)) {
                b++;
                System.out.println(a * b);

            }
        }
    }
